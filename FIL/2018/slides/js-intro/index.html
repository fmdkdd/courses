JavaScript (JS) syntax

function mapReduce(fun) {
  var tab = [1, 2, 3, 5.12];
  var sum = 0;
  for (var i = 0; i < tab.length; ++i) {
    if (fun(tab[i]) == true)
      sum += tab[i];
  }
  return sum;
}
