Objects have one prototype

let p = { x: 10, y: 10 };
let o = {
  __proto__: p,
  y: 5
};

o.x //: 10
o.y //:  5

Objects have one prototype

let p = { x: 10, y: 10 };
let o = Object.create(p);
o.y = 5;

o.x //: 10
o.y //:  5
o.__proto__ === p //: true

Every object has a prototype

let o = {};
o.toString //: function toString
o.__proto__ === Object.prototype //: true

The prototype can be null

let o = Object.create(null);
o.__proto__ === null //: true
Object.keys(o) //: []

Prototypes form a chain

let p = {x: 1};
let o = {
  __proto__: p,
  x: 2
};

Do not use in high-performance situations

Self-reference

let o = {
  x: 0,
  move(x) {
    this.x = x;
  }
};
o.move(10);
o.x //: 10

this is an implicit argument to functions, bound to the receiver

this is dynamically bound

let o = {
  x: 0,
  move(x) { this.x = x; }
};
let p = { x: 1 };
p.move = o.move;
p.move(2);
p.x //: 2

Arrow functions bind this lexically

let o = { x: 1 };
function makeMove() {
  return (x) => { this.x = x };
}
let m = makeMove.call(o);
m(2);
o.x //: 2
this.x //: undefined

Takeaways

  • Objects delegate property lookup to their prototype
  • Object.prototype is the default
  • Prototypes form a chain
  • Prototypes may be muted, but at a performance cost
  • this is dynamically bound to the receiver
  • Arrow functions bind this lexically
