function drawLevel(ctxt, level) {
  Pacmines.clearCanvas(ctxt);

  var draw = function(obj) { obj.draw(ctxt); };
  level.tiles.forEach(draw);
}

function explodeASCIIMap(string) {
  return string
    .trim()
    .split('\n')
    .map(function(string) { return string.split(''); });
